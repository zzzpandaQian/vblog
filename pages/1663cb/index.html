支付汇总

# 一.微信h5端

# 1.微信扫码：主要是通过订单id和openid获取二维码的base64地址

# 2.支付宝扫码：需要订单id

# 3.微信拉起支付：需要配置jweixin.config，通过jweixin.invoke或者wx的chooseWxpay。

# 应注意: 必须配置jweixin.conifg且jsAPiList中必须包含对应的接口名称，无论是分享还是支付，配置完成后通过jweixin.ready(),且支付必须要求openid，所以在之前必须进行静默或费静默的用户授权来获得code，再解析code获取openid。（仅支持线上模式）
# 代码:
<template>
  <view class="page">
    <u-no-network></u-no-network>
    <u-toast ref="uToast" />
    <u-top-tips ref="uTips" navbar-height="0"></u-top-tips>
    <view class="loading" v-if="loading">
      <u-loading mode="circle" size="100" color="RED"></u-loading>
    </view>
    <view class="content">
      <view class="logo"><image mode="widthFix" src="@/static/images/logo.png" /></view>
      <view class="wxpay-wrap">
        <view class="wxpay-img"
          ><image mode="widthFix" src="@/static/images/icons/wxpay.png" alt=""
        /></view>
        <view class="wxpay-title">微信支付</view>
      </view>
      <view class="price-wrap">
        <view class="price-title">支付金额</view>
        <view class="price"
          >￥<text>{{ price }}</text></view
        >
      </view>
      <view class="pay-btn-wrap">
       <u-button
          type="warning"
          @click="confirm"
          :disabled="disabled"
          :loading="disabled"
        >
          确认支付
        </u-button>
      </view>
    </view>
  </view>
</template>
<script>
import api from "@/common/api";
import {APPINFO} from "@/common/constant.js"
import {wxApi} from '@/common/wxApi.js'
// #ifdef H5
var jweixin = require("jweixin-module");
// #endif
export default {
  data() {
    return {
      loading: false,
      hash_id: "",
      price: "",
      disabled: false,
      shareid: '',
      openid: '',
    };
  },
  onShow() {
    // #ifdef H5
    this.price = uni.getStorageSync('price')
    // 获取orderid
    this.hash_id = uni.getStorageSync('order_hash_id')
    
    this.openid = uni.getStorageSync('openid')
    window.alert(this.openid)
    // 通过window.navigator.userAgent可以判断用户的手机型号
    // 或者通过uni-app的this.$u.sys 获取手机的相关信息
    if (this.$wechat && this.$wechat.isWechat()) {
      // 捕捉微信回传的code
      if(!this.openid){
        if (wxApi.getUrlParams().code != undefined) {
          this.getToken(wxApi.getUrlParams().code);
        } else {
          // 进行静默授权
          this.$login.auth();
        }
      }
      // config的请求和配置写在share函数中(通过修改jsAPiList可以使分享和支付的请求在同一个config)
      // 必须进行config否则无效，详细的share()函数见分享汇总
      this.$wechat.share()
    } else {
      uni.redirectTo({
        url: "/pages/login/index"
      });
    }
    // #endif
  },

  methods: {
    // #ifdef H5
    async getToken(code) {
      // 将获取的code解析为openid同时返回token和openid
      let request = await api.getToken({
        query: { code: code }, method: 'POST'
      });
      if (request.statusCode === 200) {
          this.openid = request.data.data.openid
      }else{
        this.$refs.uTips.show({
          title: request.data.message,
          type: "warning",
          duration: 1500
        })
        setTimeout(function() {
            uni.navigateTo({
              url: "/pages/pay/online"
            })
        }, 1500);
      }
    },
    // 拉起支付
    async confirm() {
      this.disabled = true;
      let _self = this;
      // 请求参数包含订单id和openid
      let params = {query:{order_hash_id: _self.hash_id, openid: _self.openid}, method: "POST"}
      let request = await api.pay(params);
      if (request.statusCode === 200) {
        if (request.data.status == "success") {
          jweixin.ready(function() {
            // 设定定时器是为了等待jweixin的ready跑完，安卓可能会有延迟但ios不存在该问题
            setTimeout(()=>{
              jweixin.invoke(
                // 进行支付jssdk，getBrandWCPayRequest应写在config的jsApiList里进行配置否则无效
                "getBrandWCPayRequest",
                {
                  appId: request.data.data.appId,
                  nonceStr: request.data.data.nonceStr,
                  package: request.data.data.package,
                  paySign: request.data.data.paySign,
                  signType: request.data.data.signType,
                  timeStamp: request.data.data.timeStamp
                },
                async function(res) {
                  console.log('aaa')
                  if (res.err_msg == "get_brand_wcpay_request:ok") {
                    // 跳转到支付成功页面
                    uni.redirectTo({
                      url: "/pages/index/detail?hash_id="+_self.hash_id
                    });
                    _self.disabled = false;
                  } else if (res.err_msg == "get_brand_wcpay_request:cancel") {
                    _self.$refs.uTips.show({
                      title: "支付取消",
                      type: "error",
                      duration: "2300"
                    });
                    _self.disabled = false;
                  } else {
                    _self.$refs.uTips.show({
                      title: "支付失败",
                      type: "error",
                      duration: "2300"
                    });
                    _self.disabled = false;
                  }
                }
              );
            }, 500)

          });
        } else {
          this.disabled = false;
          this.$refs.uTips.show({
            title: request.data.message,
            type: "error",
            duration: "2300"
          });
        }
      } else {
        this.disabled = false;
        this.$refs.uTips.show({
          title: request.data.message,
          type: "error",
          duration: "2300"
        });
      }
    },
    // #endif
  }
};
</script>

# 5.以下是详细的支付代码(uni-app平台)包括app端，来自github的hello-uniapp项目

<template>
	<view>
		<page-head :title="title"></page-head>
		<view class="uni-padding-wrap">
			<view style="background:#FFF; padding:50rpx 0;">
				<view class="uni-hello-text uni-center"><text>支付金额</text></view>
				<view class="uni-h1 uni-center uni-common-mt">
					<text class="rmbLogo">￥</text>
					<input class="price" type="digit" :value="price" maxlength="4" @input="priceChange" />
				</view>
			</view>
			<view class="uni-btn-v uni-common-mt">
				<!-- #ifdef MP-WEIXIN -->
				<button type="primary" @click="weixinPay" :loading="loading">微信支付</button>
				<!-- #endif -->
				<!-- #ifdef APP-PLUS -->
				<template v-if="providerList.length > 0">
					<button v-for="(item,index) in providerList" :key="index" @click="requestPayment(item,index)" :loading="item.loading">{{item.name}}支付</button>
				</template>
				<!-- #endif -->
			</view>
		</view>
	</view>
</template>
<script>
	export default {
		data() {
			return {
				title: 'request-payment',
				loading: false,
				price: 1,
				providerList: []
			}
		},
		onLoad: function() {
			// #ifdef APP-PLUS
			uni.getProvider({
				service: "payment",
				success: (e) => {
					console.log("payment success:" + JSON.stringify(e));
					let providerList = [];
					e.provider.map((value) => {
						switch (value) {
							case 'alipay':
								providerList.push({
									name: '支付宝',
									id: value,
									loading: false
								});
								break;
							case 'wxpay':
								providerList.push({
									name: '微信',
									id: value,
									loading: false
								});
								break;
							default:
								break;
						}
					})
					this.providerList = providerList;
				},
				fail: (e) => {
					console.log("获取支付通道失败：", e);
				}
			});
			// #endif
		},
		methods: {
			loginMpWeixin() {
				return new Promise((resolve, reject) => {
					uni.login({
						provider: 'weixin',
						success(res) {
							uni.request({
								url: 'https://97fca9f2-41f6-449f-a35e-3f135d4c3875.bspapp.com/http/user-center',
								method: 'POST',
								data: {
									action: 'loginByWeixin',
									params: {
										code: res.code,
										platform: 'mp-weixin'
									}
								},
								success(res) {
									if (res.data.code !== 0) {
										reject(new Error('获取openid失败：', res.result.msg))
										return
									}
									uni.setStorageSync('openid', res.data.openid)
									resolve(res.data.openid)
								},
								fail(err) {
									reject(new Error('获取openid失败：' + err))
								}
							})
						},
						fail(err) {
							reject(err)
						}
					})
				})
			},
			async weixinPay() {
				let openid = uni.getStorageSync('openid')
				console.log("发起支付");
				this.loading = true;
				if (!openid) {
					try {
						openid = await this.loginMpWeixin()
					} catch (e) {
						console.error(e)
					}
					if (!openid) {
						uni.showModal({
							content: '获取openid失败',
							showCancel: false
						})
						this.loading = false
						return
					}
				}
				this.openid = openid
				let orderInfo = await this.getOrderInfo('wxpay')
				if (!orderInfo) {
					uni.showModal({
						content: '获取支付信息失败',
						showCancel: false
					})
					return
				}
				uni.requestPayment({
					...orderInfo,
					success: (res) => {
						uni.showToast({
							title: "感谢您的赞助!"
						})
					},
					fail: (res) => {
						uni.showModal({
							content: "支付失败,原因为: " + res
								.errMsg,
							showCancel: false
						})
					},
					complete: () => {
						this.loading = false;
					}
				})
			},
			async requestPayment(e, index) {
				this.providerList[index].loading = true;
				const provider = e.id
				let orderInfo = await this.getOrderInfo(provider);
				if (!orderInfo) {
					uni.showModal({
						content: '获取支付信息失败',
						showCancel: false
					})
					return
				}
				console.log('--------orderInfo--------')
				console.log(orderInfo);
				uni.requestPayment({
					provider,
					orderInfo: orderInfo,
					success: (e) => {
						console.log("success", e);
						uni.showToast({
							title: "感谢您的赞助!"
						})
					},
					fail: (e) => {
						console.log("fail", e);
						uni.showModal({
							content: "支付失败,原因为: " + e.errMsg,
							showCancel: false
						})
					},
					complete: () => {
						this.providerList[index].loading = false;
					}
				})
			},
			getOrderInfo(provider) {
				return new Promise((resolve, reject) => {
					if (!this.price) {
						reject(new Error('请输入金额'))
					}
					uni.request({
						method: 'POST',
						url: 'https://97fca9f2-41f6-449f-a35e-3f135d4c3875.bspapp.com/http/pay',
						data: {
							provider,
							openid: this.openid,
							totalFee: Number(this.price) * 100, // 转为以分为单位
							// #ifdef APP-PLUS
							platform: 'app-plus',
							// #endif
							// #ifdef MP-WEIXIN
							platform: 'mp-weixin',
							// #endif
						},
						success(res) {
							if (res.data.code === 0) {
								resolve(res.data.orderInfo)
							} else {
								reject(new Error('获取支付信息失败' + res.data.msg))
							}
						},
						fail(err) {
							reject(new Error('请求支付接口失败' + err))
						}
					})
				})
			},
			priceChange(e) {
				console.log(e.detail.value)
				this.price = e.detail.value;
			}
		}
	}
</script>
